#include<iostream>
using namespace std;
void swap(int *a, int *b)
{
    int temp = *a;
    *a = *b;
    *b = temp;
}
int main()
{
    int SIZE;
    cin>> SIZE;
    int arr[SIZE];
    for(int i = 0;i < SIZE; i++)
	cin>> arr[i];
    for(int i = 0; i < SIZE-1; i++)
    {
	for(int j = i; j >= 0; j--)
	{
	    if(arr[j] > arr[j+1])
	        swap( &arr[j], &arr[j+1]);
	    else
	        break;
	}
    }
    for(int i = 0; i < SIZE; i++)
	cout<< arr[i]<<" ";
    return 0;
}
